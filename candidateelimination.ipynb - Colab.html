    n = X.shape[1]
    # Initialize S and G

                elif S[j] != X[i][j]:
                    S[j] = '?'
            # Remove hypotheses from G inconsistent with positive example

        else:   # Negative example
            new_G = []

                    for j in range(n):
                        if S[j] != '?' and S[j] != '' and S[j] != X[i][j]:
                            new_h = g.copy()
                            new_h[j] = S[j]
